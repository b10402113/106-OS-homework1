Chapter 2 - Page tables
作業系統通過page tables機制對內存空間的控制。使得xv6能夠讓不同processes各自的地址空間用到單一的physical memory，還能夠為不同processes的記憶體提供保護。xv6主要利用page tables來區分多個地址空間和保護記憶體。它也使用了一些簡單的page-table技巧，mapping同樣的記憶體(核心) 到多的地址空間，mapping同樣的記憶體多次到同一個地址空間(使用者的每個page都會用到內核部分)，以及一個沒有用到的page保護一位使用者的stack。x86指令都是計算virtual address(使用者和內核都是), 一個x86 page table就是一個2^20 (1,048,576) page table entries (PTEs)的陣列，每條PTE包含了一個20-bit的physical page number(PPN)和一些flags，說明Paging hardware對應的virtual address的使用權限。PTE_P表示PTE是否陳列在page tables中。PTE_W控制能不能對該頁寫指令。PTE_U控制使用者program能否使用該頁。這些的flags和所有page hardware相關的指令都在mmu.h（0700）定義。
建立一個地址空間，main呼叫kvmalloc，創建並轉換到一個內核運行在KERNBASE以上所需配對的page table。
kvmalloc(void)
{
kpgdir = setupkvm();
switchkvm();
}
大多數工作都是由setupkvm實作的，setupkvm不會建立任何使用者內存的mapping。
setupkvm(void)
{
pde_t pgdir;
struct kmap k;
if((pgdir = (pde_t)kalloc()) == 0)
return 0;
memset(pgdir, 0, PGSIZE);
if (p2v(PHYSTOP) > (void)DEVSPACE)
panic(“PHYSTOP too high”);
for(k = kmap; k < &kmap[NELEM(kmap)]; k++)
if(mappages(pgdir, k−>virt, k−>phys_end − k−>phys_start,
(uint)k−>phys_start, k−>perm) < 0)
return 0;
return pgdir;
}
首先它會分配一頁內存來放置page directory，然後呼叫mappages建立內核需要的mapping，這些mapping的數據會在kap中找到。
kmap[] = {
{ (void*)KERNBASE, 0, EXTMEM, PTE_W}, // I/O space
{ (void*)KERNLINK, V2P(KERNLINK), V2P(data), 0}, // kern text+rodata
{ (void*)data, V2P(data), PHYSTOP, PTE_W}, // kern data+memory
{ (void*)DEVSPACE, DEVSPACE, 0, PTE_W}, // more devices
};
mappages是在page table建立一段virtual addresses到一段physical addresses的mapping。對於每一個等待mapping的virtual address，mappages呼叫walkpgdir來找到該地址對應的PTE地址。
mappages(pde_t *pgdir, void *va, uint size, uint pa, int perm)
{
char *a, *last;
pte_t *pte;

a = (char*)PGROUNDDOWN((uint)va);
last = (char*)PGROUNDDOWN(((uint)va) + size − 1);
for(;;){
if((pte = walkpgdir(pgdir, a, 1)) == 0)
return −1;
if(*pte & PTE_P)
panic(“remap”);
*pte = pa | perm | PTE_P; //初始化該PTE以保存對應physical page number、許可的權限（PTE_W和/或PTE_U）以及PTE_P來mark該PTE是否是有效的。
if(a == last)
break;
a += PGSIZE;
pa += PGSIZE;
}
return 0;
}
walkpgdir模擬x86的page hardware為一個virtual address的動作。
walkpgdir(pde_t *pgdir, const void *va, int alloc)
{
pde_t *pde;
pte_t *pgtab;

pde = &pgdir[PDX(va)]; //walkpgdir通過virtual address的前10位來找到在page directory的對應條目，如果該條目不存在，說明要找的page table pag尚未分配。
if(pde & PTE_P){
pgtab = (pte_t)p2v(PTE_ADDR(pde));
}
else {
if(!alloc || (pgtab = (pte_t)kalloc()) == 0)//如果alloc參數被設置了，walkpgdir會分配page table page將其physical address放到page directory。
return 0;
// Make sure all those PTE_P bits are zero.
memset(pgtab, 0, PGSIZE);
// The permissions here are overly generous, but they can
// be further restricted by the permissions in the page table
// entries, if necessary.
*pde = v2p(pgtab) | PTE_P | PTE_W | PTE_U;
}
return &pgtab[PTX(va)]; //最後用virtual address的接下來10位來找到其在page table的PTE地址
}
Physical memory allocation，在運行時，內核需要為page table、process的使用者內存、內核stack及管道buffers分配空閒的physical內存。分配器的資料結構是一個由可分配physical內存頁構成的free list。每個free page’s list的元素是struct run。
struct run {
struct run *next;
};
main呼叫kinit1和kinit2兩個函數對分配器進行初始化。kinit1 and kinit2 呼叫freerange增加內存到空間列表，freerange對每一頁通過呼叫kfree來實現。
kinit1(void *vstart, void *vend)
{
initlock(&kmem.lock, “kmem”);
kmem.use_lock = 0;
freerange(vstart, vend);
}
kinit2(void *vstart, void *vend)
{
freerange(vstart, vend);
kmem.use_lock = 1;
}
函數kfree首先將被釋放內存的每一byte設為1，造成代碼所讀到的不是原有數據，而是垃圾數據，能讓這種錯誤的代碼盡快壞掉。
kfree(char *v)
{
struct run *r;

if((uint)v % PGSIZE || v < end || v2p(v) >= PHYSTOP)
panic(“kfree”);

// Fill with junk to catch dangling refs.
memset(v, 1, PGSIZE);

if(kmem.use_lock)
acquire(&kmem.lock);
r = (struct run*)v; //接下來kfree把v轉換為一個指向struct run的指針r
r−>next = kmem.freelist; //在r->next中保存原有空閒列表的表頭
kmem.freelist = r; //將當前的空閒列表設置為r
if(kmem.use_lock)
release(&kmem.lock);
}
sbrk是一個system call用來縮小或增加process的內存。sbrk是由growproc函數實現。
growproc(int n)
{
uint sz;
struct proc *curproc = myproc();

sz = curproc−>sz;
if(n > 0){ //n為正，分配一個或多個physical pages，並mapping到process地址空間的頂部
if((sz = allocuvm(curproc−>pgdir, sz, sz + n)) == 0)
return −1;
} else if(n < 0){ //n為負，取消mapping到process地址空間，且釋放對應的physical pages
if((sz = deallocuvm(curproc−>pgdir, sz, sz + n)) == 0)
return −1;
}
curproc−>sz = sz;
switchuvm(curproc);
return 0;
}
exec是創建地址空間中使用者部分的system call。它根據file system中保存的某個文件來初始化使用者部分。
exec(char *path, char **argv)
{
char *s, *last;
int i, off;
uint argc, sz, sp, ustack[3+MAXARG+1];
struct elfhdr elf; //一個ELF二進制文件包括了一個ELF頭
struct inode *ip;
struct proghdr ph; //連續幾個程序段的頭
pde_t *pgdir, *oldpgdir;
struct proc *curproc = myproc();

begin_op();

if((ip = namei(path)) == 0){ //exec通過namei打開二進制文件。
end_op();
cprintf(“exec: fail\n”);
return −1;
}
ilock(ip);
pgdir = 0;

// Check ELF header
if(readi(ip, (char*)&elf, 0, sizeof(elf)) != sizeof(elf))
goto bad;
if(elf.magic != ELF_MAGIC) //如果ELF頭中包含正確的魔法數字，exec就會認為該二ㄑ進制文件是正確的。
goto bad;

if((pgdir = setupkvm()) == 0)//exec通過setupkvm分配了一個沒有使用者mapping的page table
goto bad;

// Load program into memory.
sz = 0;
for(i=0, off=elf.phoff; i<elf.phnum; i++, off+=sizeof(ph)){
if(readi(ip, (char*)&ph, off, sizeof(ph)) != sizeof(ph))
goto bad;
if(ph.type != ELF_PROG_LOAD)
continue;
if(ph.memsz < ph.filesz)
goto bad;
if(ph.vaddr + ph.memsz < ph.vaddr)
goto bad;
if((sz = allocuvm(pgdir, sz, ph.vaddr + ph.memsz)) == 0) //通過allocuvm為每個ELF段分配內存
goto bad;
if(ph.vaddr % PGSIZE != 0)
goto bad;
if(loaduvm(pgdir, (char*)ph.vaddr, ip, ph.off, ph.filesz) < 0) //通過loaduvm把段的內容載入內存中
goto bad;
}
iunlockput(ip);
end_op();
ip = 0;

// Allocate two pages at the next page boundary.
// Make the first inaccessible. Use the second as the user stack.
sz = PGROUNDUP(sz);
if((sz = allocuvm(pgdir, sz, sz + 2PGSIZE)) == 0)
goto bad;
clearpteu(pgdir, (char)(sz − 2*PGSIZE));
sp = sz;

// Push argument strings, prepare rest of stack in ustack.
for(argc = 0; argv[argc]; argc++) {
if(argc >= MAXARG)
goto bad;
sp = (sp − (strlen(argv[argc]) + 1)) & ~3;
if(copyout(pgdir, sp, argv[argc], strlen(argv[argc]) + 1) < 0)
goto bad;
ustack[3+argc] = sp;
}
ustack[3+argc] = 0;

ustack[0] = 0xffffffff; // fake return PC
ustack[1] = argc;
ustack[2] = sp − (argc+1)*4; // argv pointer

sp −= (3+argc+1) * 4;
if(copyout(pgdir, sp, ustack, (3+argc+1)*4) < 0)
goto bad;

// Save program name for debugging.
for(last=s=path; *s; s++)
if(*s == ’/’)
last = s+1;
safestrcpy(curproc−>name, last, sizeof(curproc−>name));

// Commit to the user image.
oldpgdir = curproc−>pgdir;
curproc−>pgdir = pgdir;
curproc−>sz = sz;
curproc−>tf−>eip = elf.entry; // main
curproc−>tf−>esp = sp;
switchuvm(curproc); //一旦新的內存影像建立完成，exec就能裝載新影像
freevm(oldpgdir); //把舊影像釋放
return 0; //exec成功地返回0

bad:
if(pgdir)
freevm(pgdir);
if(ip){
iunlockput(ip);
end_op();
}
return −1;
}
