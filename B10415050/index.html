---------Code: creating an address space----------

main呼叫kvmalloc，創建並切換到一個擁有內核運行所需的KERNBASE以上映射(mapping)的頁表。這裡的大多數工作都是由setupkvm完成的。首先，它會分配一頁內存(user memory)來放置頁目錄(page directory)，然後調用mappages來建立內核需要的映射，這些映射可以在kmap數組中找到。這裡的映射包括內核的指令和數據，PHYSTOP以下的物理內存(physical memory)，以及I/O設備所佔的內存。setupkvm不會建立任何用戶內存的映射，這些映射稍後會建立。
// #define PTE_U 0x004 // User
#define PHYSTOP 0xE000000 // Top physical memory

kvmalloc(void)// Allocate one page table for the machine for the kernel address
 {            // space for scheduler processes
    kpgdir = setupkvm(); 
    switchkvm(); //呼叫switchkvm
 }

 setupkvm(void)// Set up kernel part of a page table
 {
     pde_t *pgdir;
     struct kmap *k; //pointer 按照kmap data structure 映射 virtual address to physical address

     if((pgdir = (pde_t*)kalloc()) == 0) //一樣則return 0
         return 0;
     memset(pgdir, 0, PGSIZE);
     if (P2V(PHYSTOP) > (void*)DEVSPACE)
         panic("PHYSTOP too high");
     for(k = kmap; k < &kmap[NELEM(kmap)]; k++)
         if(mappages(pgdir, k−>virt, k−>phys_end − k−>phys_start,
             (uint)k−>phys_start, k−>perm) < 0) {
         freevm(pgdir); //清空
     return 0;
    }
     return pgdir;
 }

 // This table defines the kernel’s mappings, which are present in
 // every process’s page table.
 static struct kmap {
 void *virt;
 uint phys_start;
 uint phys_end;
 int perm;
 } kmap[] = {
 { (void*)KERNBASE, 0, EXTMEM, PTE_W}, // I/O space
 { (void*)KERNLINK, V2P(KERNLINK), V2P(data), 0}, // kern text+rodata
 { (void*)data, V2P(data), PHYSTOP, PTE_W}, // kern data+memory
 { (void*)DEVSPACE, DEVSPACE, 0, PTE_W}, // more devices
};

mappages做的工作是在頁表中建立一段虛擬內存(vitual memory)到一段物理內存的映射。它是在頁的級別(level of pages)，即一頁一頁地建立映射的。對於每一個待映射虛擬地址，mappages調用walkpgdir來找到該地址對應的PTE地址。然後初始化該PTE以保存對應物理頁號、許可級別（PTE_W和/或PTE_U）以及PTE_P位來標記該PTE是否是有效的。

// Create PTEs for virtual addresses starting at va that refer to
// physical addresses starting at pa. va and size might not
// be page−aligned.
mappages(pde_t *pgdir, void *va, uint size, uint pa, int perm)
{ 
    char *a, *last //地址的啟始端與結束端
    pte_t *pte;

    a = (char*)PGROUNDDOWN((uint)va);
    last = (char*)PGROUNDDOWN(((uint)va) + size − 1);
    for(;;){
        if((pte = walkpgdir(pgdir, a, 1)) == 0) //一開始把 a 跟 last 分別對齊其頁的初始地址；進入迴圈後呼叫 walkpgdir()
            return −1;
         if(*pte & PTE_P)
             panic("remap");
     *pte = pa | perm | PTE_P   // Set 權限及將 PTE_P 標為 valid
     if(a == last)
         break;
     a += PGSIZE;
     pa += PGSIZE;
     }
 return 0;
}

walkpgdir模仿x86的分頁硬件為一個虛擬地址尋找PTE的過程。walkpgdir通過虛擬地址的前10位元來找到在頁目錄中的對應條，如果該條目不存在，說明要找的頁表頁尚未分配；如果alloc參數被設置了，walkpgdir會分配頁表頁並將其物理地址放到頁目錄中。最後用虛擬地址的接下來10位元來找到其在頁表中的PTE地址。

// Return the address of the PTE in page table pgdir
// that corresponds to virtual address va. If alloc!=0,
// create any required page table pages.
static pte_t *
//先找高位元 10 bit 的 pageDirectory，再找後 10 bit 的 pageTable
walkpgdir(pde_t *pgdir, const void *va, int alloc)
{
    pde_t *pde;
    pte_t *pgtab;

    pde = &pgdir[PDX(va)];
     if(*pde & PTE_P){
         pgtab = (pte_t*)P2V(PTE_ADDR(*pde));
     } else {
     if(!alloc || (pgtab = (pte_t*)kalloc()) == 0)
         return 0;
     // Make sure all those PTE_P bits are zero.
     memset(pgtab, 0, PGSIZE);
     // The permissions here are overly generous, but they can
     // be further restricted by the permissions in the page table
     // entries, if necessary.
     *pde = V2P(pgtab) | PTE_P | PTE_W | PTE_U;
 }
    return &pgtab[PTX(va)];
 }

--------Code: Physical memory allocator-----------

allcator中的資料結構是一個由可分配物理內存頁構成的空閒鍊錶(free list)。這個空閒頁的鍊錶的元素是結構體struct run。那麼分配器從哪裡獲得內存來存放這些數據結構呢？實際上，分配器將每個空閒頁的run結構體保存在該空閒頁本身中，因為空閒頁中沒有其他數據。分配器還用一個spin lock來保護空閒鍊錶。鍊錶和這個鎖都封裝在一個結構體中，這樣邏輯就比較明晰：鎖保護了該結構體中的域。不過現在讓我們先忽略這個鎖，以及對acquire和release的調用；我們會在第4章了解其細節。

 struct run {
 struct run *next; //Link list
 };

// Initialization happens in two phases.
// 1. main() calls kinit1() while still using entrypgdir to place just
// the pages mapped by entrypgdir on free list.
// 2. main() calls kinit2() with the rest of the physical pages
// after installing a full page table that maps them on all cores.
void
kinit1(void *vstart, void *vend)
 {
 initlock(&kmem.lock, "kmem");
 kmem.use_lock = 0;
 freerange(vstart, vend);
 }

 kinit2(void *vstart, void *vend)
 {
freerange(vstart, vend);
 kmem.use_lock = 1;
 }

main函數調用了kinit1和kinit2兩個函數對分配器進行初始化。這樣做是由於main中的大部分代碼都不能使用鎖以及4MB以上的內存。kinit1在前4MB進行了不需要鎖的內存分配。而kinit2允許了鎖的使用，並使得更多的內存可用於分配。原本應該由main決定有多少物理內存可用於分配，但在x86上很難實現。所以它假設機器中有240MB（PHYSTOP）物理內存，並將內核末尾和PHYSTOP之間的內存都作為一個初始的空閒內存池。kinit1和kinit2調用freerange將內存加入空閒鍊錶中，freerange則是通過對每一頁調用kfree實現該功能。一個PTE只能指向一個4096字節對齊的物理地址（即是4096的倍數），因此freerange用PGROUNDUP來保證分配器只會釋放對齊的物理地址。分配器原本一開始沒有內存可用，正是對kfree的調用將可用內存交給了分配器來管理。

 #define PGROUNDUP(sz) (((sz)+PGSIZE−1) & ~(PGSIZE−1))

 freerange(void *vstart, void *vend)
 {
 char *p;
 p = (char*)PGROUNDUP((uint)vstart);
 for(; p + PGSIZE <= (char*)vend; p += PGSIZE)
 kfree(p);
 }

 // Free the page of physical memory pointed at by v,
 // which normally should have been returned by a
 // call to kalloc(). (The exception is when
 // initializing the allocator; see kinit above.)
 void
 kfree(char *v)
 {
 struct run *r;

 if((uint)v % PGSIZE || v < end || V2P(v) >= PHYSTOP)
 panic("kfree");

 // Fill with junk to catch dangling refs.
 memset(v, 1, PGSIZE);

 if(kmem.use_lock)
 acquire(&kmem.lock);
 r = (struct run*)v;
 r−>next = kmem.freelist;
 kmem.freelist = r;
 if(kmem.use_lock)
 release(&kmem.lock);
 }

 // Allocate one 4096−byte page of physical memory.
 // Returns a pointer that the kernel can use.
 // Returns 0 if the memory cannot be allocated.
 char*
 kalloc(void)
 {
 struct run *r;

 if(kmem.use_lock)
 acquire(&kmem.lock);
 r = kmem.freelist;
 if(r)
 kmem.freelist = r−>next;
 if(kmem.use_lock)
 release(&kmem.lock);
 return (char*)r;
 }

分配器用映射到高內存區域的虛擬地址找到對應的物理頁，而非物理地址。所以kinit會使用p2v(PHYSTOP)來將PHYSTOP（一個物理地址）翻譯為虛擬地址。分配器有時將地址看作是整型，這是為了對其進行運算（譬如在kinit中遍歷所有頁）；而有時將地址看作讀寫內存用的指針（譬如操作每個頁中的run結構體） ；對地址的雙重使用導致分配器代碼中充滿了類型轉換。另外一個原因是，釋放和分配內存隱性地改變了內存的類型。

函數kfree首先將被釋放內存的每一字節設為1。這使得訪問已被釋放內存的代碼所讀到的不是原有數據，而是垃圾數據；這樣做能讓這種錯誤的代碼儘早崩潰。接下來kfree把v轉換為一個指向結構體struct run的指針，在r->next中保存原有空閒鍊錶的表頭，然後將當前的空閒鍊錶設置為r。kalloc移除並返回空閒鍊錶的表頭。

Chapter 2的圖2-3展示了在xv6中，一個運行中進程的用戶內存結構。堆在stack之上，所以它可以增長（通過sbrk）。stack佔用了單獨的一頁內存，其中存有exec創建的初始數據。stack的最上方放著字符串形式的命令行參數以及指向這些參數的指針數組，其下方放的值使得一個程序可以從main開始，彷彿剛剛調用了函數main(argc, argv)。為了防止stack使用了它不應該使用的頁，stack的下方有一個保護頁。保護頁沒有映射，因此當stack的增長超出其所在頁時就會產生異常，因為無法翻譯這個錯誤的地址。

-----Code: sbrk-----

Sbrk是一個system call用來縮小或增加其內存。由函數growproc（2558）實現。如果n是postive，growproc分配一個或更多的物理頁面，並將它們映射到進程地址空間的頂部。如果n是negative，growproc會從進程的地址空間中取消映射一個或多個頁面釋放相應的物理頁面。要進行這些更改，xv6會修改進程頁表。進程的頁表存儲在內存中，因此內核可以用普通的賦值語句更新表，這就是allocuvm和deallocuvm做的事。 x86硬件緩存翻譯中的頁表條目Look-aside緩衝區（TLB），當xv6更改頁表時，它必須無效緩存的條目。如果它沒有使緩存的條目無效，那麼在某個時間點TLB可能會使用舊的映射，指向同一時間的物理頁面已分配給另一個進程，結果是一個進程可能會抹掉在一些其他進程的記憶。 Xv6通過重新加載使失效的緩存條目無效cr3，保存當前頁表地址的寄存器。

 sys_sbrk(void)
 {
 int addr;
 int n;

 if(argint(0, &n) < 0)
 return −1;
 addr = myproc()−>sz;
 if(growproc(n) < 0)
 return −1;
 return addr;
 }

-----Code: exec-----

exec是創建地址空間中用戶部分的系統調用。它根據文件系統中保存的某個文件來初始化用戶部分。exec通過namei打開二進製文件，這一點將在第6章進行解釋。然後，它讀取ELF頭。xv6應用程序以通行的ELF格式來描述，該格式在elf.h中定義。一個ELF二進製文件包括了一個ELF頭，即結構體struct elfhdr（0955），然後是連續幾個程序段的頭，即結構體struct proghdr。每個proghdr都描述了需要載入到內存中的程序段。xv6中的程序只有一個程序段的頭，但其他操作系統中可能有多個。

exec第一步是檢查文件是否包含ELF二進制代碼。一個ELF二進製文件是以4個“魔法數字”開頭的，即0x7F，“E”，“L”，“F”，或者寫為宏ELF_MAGIC。如果ELF頭中包含正確的魔法數字，exec就會認為該二進製文件的結構是正確的。

exec通過setupkvm分配了一個沒有用戶部分映射的頁表，再通過allocuvm為每個ELF段分配內存，然後通過loaduvm把段的內容載入內存中。allocuvm會檢查請求分配的虛擬地址是否是在KERNBASE之下。loaduvm通過walkpgdir來找到寫入ELF段的內存的物理地址；通過readi來將段的內容從文件中讀出。

exec創建的第一個用戶程序/init程序段的頭是這樣的：

#objdump -p _init

_init: file format elf32-i386

Program Header:
LOAD off 0x00000054 vaddr 0x00000000 paddr 0x00000000 align 2**2
filesz 0x000008c0 memsz 0x000008cc flags
filesz可能比memsz小，這表示中間相差的地方應該用0填充（對於C的全局變量）而不是繼續從文件中讀數據。對於/init，filesz是2240字節而memsz是2252字節。所以allocuvm會分配足夠的內存來裝2252字節的內容，但只從文件/init中讀取2240字節的內容。

現在exec要分配以及初始化用戶棧了。它只為棧分配一頁內存。exec一次性把參數字符串拷貝到棧頂，然後把指向它們的指針保存在ustack中。它還會在main參數列表argv的最後放一個空指針。這樣，ustack中的前三項就是偽造的返回PC，argc和argv指針了。

exec會在stack的頁下方放一個無法訪問的頁，這樣當程序嘗試使用超過一個頁的stack時就會出錯。另外，這個無法訪問的頁也讓exec能夠處理那些過於龐大的參數；當參數過於龐大時，exec用於將參數拷貝到棧上的函數copyout會發現目標頁無法訪問，並且return -1。

在創建新的內存映像時，如果exec發現了錯誤，比如一個無效的程序段，它就會跳轉到標記bad處，釋放這段內存映像，然後返回-1。exec必須在確認該調用可以成功後才能釋放原來的內存映像，否則，若原來的內存映像被釋放，exec甚至都無法向它return -1了。exec中的錯誤只可能發生在新建內存映像時。一旦新的內存映像建立完成，exec就能裝載新映像而把舊映像釋放。最後，exec成功地return 0。

 // Allocate page tables and physical memory to grow process from oldsz to
 // newsz, which need not be page aligned. Returns new size or 0 on error.
 int
 allocuvm(pde_t *pgdir, uint oldsz, uint newsz)
 {
     char *mem;
     uint a;

     if(newsz >= KERNBASE) //KERNBASE以下才能使用
        return 0;
     if(newsz < oldsz)
        return oldsz;

     a = PGROUNDUP(oldsz);
     for(; a < newsz; a += PGSIZE){
         mem = kalloc();
     if(mem == 0){
         cprintf("allocuvm out of memory\n");
     deallocuvm(pgdir, newsz, oldsz);
     return 0;
 }
     memset(mem, 0, PGSIZE);
     if(mappages(pgdir, (char*)a, PGSIZE, V2P(mem), PTE_W|PTE_U) < 0){
        cprintf("allocuvm out of memory (2)\n"); deallocuvm(pgdir, newsz, oldsz);
     kfree(mem);


     return 0;
 }
 }
 return newsz;
 }

 // Load a program segment into pgdir. addr must be page−aligned
 // and the pages from addr to addr+sz must already be mapped.
 int
loaduvm(pde_t *pgdir, char *addr, struct inode *ip, uint offset, uint sz)
 {
uint i, pa, n;
 pte_t *pte;

 if((uint) addr % PGSIZE != 0)
 panic("loaduvm: addr must be page aligned");
for(i = 0; i < sz; i += PGSIZE){
if((pte = walkpgdir(pgdir, addr+i, 0)) == 0)
 panic("loaduvm: address should exist");
 pa = PTE_ADDR(*pte);
 if(sz − i < PGSIZE)
 n = sz − i;
 else
 n = PGSIZE;
 if(readi(ip, P2V(pa), offset+i, n) != n)
 return −1;
 }
 return 0;
 }

 // Read data from inode.
 // Caller must hold ip−>lock.
 int
 readi(struct inode *ip, char *dst, uint off, uint n)
 {
     uint tot, m;
     struct buf *bp;

     if(ip−>type == T_DEV){
     if(ip−>major < 0 || ip−>major >= NDEV || !devsw[ip−>major].read)
     return −1;
     return devsw[ip−>major].read(ip, dst, n);
     }

    if(off > ip−>size || off + n < off)
    return −1;
    if(off + n > ip−>size)
    n = ip−>size − off;

    for(tot=0; tot<n; tot+=m, off+=m, dst+=m){
    bp = bread(ip−>dev, bmap(ip, off/BSIZE));
    m = min(n − tot, BSIZE − off%BSIZE);
    memmove(dst, bp−>data + off%BSIZE, m);
     brelse(bp);
 }
return n;
}

// Copy len bytes from p to user address va in page table pgdir.
// Most useful when pgdir is not the current page table.
// uva2ka ensures this only works for PTE_U pages.
 int
 copyout(pde_t *pgdir, uint va, void *p, uint len)
 {
     char *buf, *pa0;
     uint n, va0;

     buf = (char*)p;
     while(len > 0){
     va0 = (uint)PGROUNDDOWN(va);
     pa0 = uva2ka(pgdir, (char*)va0);
     if(pa0 == 0)
         return −1;
     n = PGSIZE − (va − va0);
    if(n > len)
         n = len;
    memmove(pa0 + (va − va0), buf, n);
    len −= n;
     buf += n;
     va = va0 + PGSIZE;
     }
 return 0;
 }
