Chapter 2 - Page table

運作原理

1. Paging hardware

將原本的virtual address分成三個部分

  • Page directory(10bit)：頁表目錄
  • Page table(10bit)：頁表
  • Offset(12bit)

step1. 利用前10bit在Page directory中找到對應的PTE，此PTE會指向一個對應的Page table
step2. 利用後10bit找到Page table中對應的PTE
step3. 此PTE的PPN就是轉換後Physical address的前20bit，後12bit則是原本的Offset。

  • Page directory有2^10個Page table，每個Page table中有2^10個PTE

  • PTE：page table entrie，包含20bit的PPN（physical page number）和Flag

  • PTE的flag bits各代表不同的意義，包含是否存在或是否可以讀寫…等。

2. Physical memory allocation

  • physical memory的allocates 和 frees都是一個4096byte的page。
  • 未使用的physical memory會以一個 linked list來管理
  • memory allocate -> 從free list移除
    memory free -> 加回free list中

3.Process address space

  • 每個process都有自己的page，切換process時也同時切換page
  • 一個process一開始可以使用的memory只有2GB（virtual address：0~KERNBASE）
  • process需要更多memory時，系統就去尋找free的physical pages
  • 設定PTEs的Flag為可使用（PTE_U, PTE_W, and PTE_P）
    • 通常physical pages中不會所有的PTE都有用到，沒被使用的PTEs的PTE_P=0
  • 此physical pages會mapping到KERNBASE之上

程式流程

1. creating an address space

程式說明（//註解）






 kvmalloc(void)
 {
     kpgdir = setupkvm();    //create page table
     switchkvm();
 }




















setupkvm(void)
{
     pde_t *pgdir;    //page directory
     struct kmap *k;

     if((pgdir = (pde_t*)kalloc()) == 0)//allocate一個page memory存放page　directory
         return 0;
     memset(pgdir, 0, PGSIZE);
     if (P2V(PHYSTOP) > (void*)DEVSPACE)
         panic("PHYSTOP too high");
     for(k = kmap; k < &kmap[NELEM(kmap)]; k++)
         //呼叫mappages來連接程式的virtual adress和分配給他的physical addresses
        if(mappages(pgdir, k−>virt, k−>phys_end − k−>phys_start,
         (uint)k−>phys_start, k−>perm) < 0) {
         freevm(pgdir);
         return 0;
     }
     return pgdir;
 }























 mappages(pde_t *pgdir, void *va, uint size, uint pa, int perm)
 {
 //這裡是一個page一個page的將virtual adress和分配給他的physical addresses連接起來
     char *a, *last;
     pte_t *pte;  //page table entrie，會有對應的 physical addresses

     a = (char*)PGROUNDDOWN((uint)va);
     last = (char*)PGROUNDDOWN(((uint)va) + size − 1);
     for(;;){
         if((pte = walkpgdir(pgdir, a, 1)) == 0)
         //呼叫walkpgdir找到這個PTE對應的physical addresses
             return −1;
         if(*pte & PTE_P)
             panic("remap");
         *pte = pa | perm | PTE_P;    //初始化這個PTE的Flag
         if(a == last)
             break;
         a += PGSIZE;
         pa += PGSIZE;
     }
     return 0;
 }























 walkpgdir(pde_t *pgdir, const void *va, int alloc)
 {
     pde_t *pde;
     pte_t *pgtab;

     pde = &pgdir[PDX(va)];    
     //取Virtual address的前10bit來找到page　directory中對應的page table
     //有找到page table且PTE有效(PTE_P=1)才能使用
     if(*pde & PTE_P){
         pgtab = (pte_t*)P2V(PTE_ADDR(*pde));
     } else {
        if(!alloc || (pgtab = (pte_t*)kalloc()) == 0)
         return 0;
         // Make sure all those PTE_P bits are zero.
        memset(pgtab, 0, PGSIZE);
         // The permissions here are overly generous,but they can
        // be further restricted by the permissions in the page table
        // entries, if necessary.
         *pde = V2P(pgtab) | PTE_P | PTE_W | PTE_U;
     }
    return &pgtab[PTX(va)];
 }

2. Physical memory allocator

程式說明（//註解）


















kinit1(void *vstart, void *vend)
{
    //kinit1負責不使用lock的前4 megabytes 的allocation
     initlock(&kmem.lock, "kmem");
     kmem.use_lock = 0;
     freerange(vstart, vend);
}
 
void
kinit2(void *vstart, void *vend)
{
    //kinit2則可以使用lock，且負責超過4 megabytes的allocation
    freerange(vstart, vend);
    kmem.use_lock = 1;
}

//kinit1和kinit2都呼叫freerange來free memory，將沒有使用的memory加到free list中










freerange(void *vstart, void *vend)
{
//freerange先利用PGROUNDUP確定只會free到對應的physical addresses
//再呼叫kfree完成剩下的工作
    char *p;
    p = (char*)PGROUNDUP((uint)vstart);
    for(; p + PGSIZE <= (char*)vend; p += PGSIZE)
        kfree(p);
 }



















kfree(char *v)
{
    struct run *r;

    if((uint)v % PGSIZE || v < end || V2P(v) >= PHYSTOP)
        panic("kfree");

    // 將要被free的memory中的資料全部設為1，清除資料
    memset(v, 1, PGSIZE);

    if(kmem.use_lock)
    acquire(&kmem.lock);
    r = (struct run*)v;
    r−>next = kmem.freelist;    //將已被free的memory加到free　list中
    kmem.freelist = r;
    if(kmem.use_lock)
        release(&kmem.lock);
 }

3.sbrk

程式說明（//註解）






















growproc(int n)
 {
 //n>0：分配更多physical pages給程式的address space
 //n<0：從程式的address space移除不需要的physical pages
     uint sz;
     struct proc *curproc = myproc();

     sz = curproc−>sz;
     if(n > 0){
     //呼叫allocuvm來存取更多physical pages
         if((sz = allocuvm(curproc−>pgdir, sz, sz + n)) == 0)
             return −1;
     } else if(n < 0){
     //呼叫deallocuvm移除不需要的physical pages
     if((sz = deallocuvm(curproc−>pgdir, sz, sz + n)) == 0)
         return −1;
     }
     curproc−>sz = sz;
     switchuvm(curproc);
     return 0;
 }

4.exec

程式說明（//註解）














































































































 exec(char *path, char **argv)
 {
     char *s, *last;
     int i, off;
     uint argc, sz, sp, ustack[3+MAXARG+1];
     struct elfhdr elf;
     struct inode *ip;
     struct proghdr ph;
     pde_t *pgdir, *oldpgdir;
     struct proc *curproc = myproc();

     begin_op();
    
     if((ip = namei(path)) == 0){
     //用namei來開啟path指的文件
         end_op();
         cprintf("exec: fail\n");
         return −1;
     }
     ilock(ip);
     pgdir = 0;

     // 檢查ELF的格式是否正確
     if(readi(ip, (char*)&elf, 0, sizeof(elf)) != sizeof(elf))
         goto bad;
     if(elf.magic != ELF_MAGIC)//MAGIC NUMBER={0x7F, ’E’, ’L’, ’F’}
         goto bad;

     if((pgdir = setupkvm()) == 0)//透過setupkvm allocate新的page table
         goto bad;

     // Load program into memory.
     sz = 0;
     for(i=0, off=elf.phoff; i<elf.phnum; i++, off+=sizeof(ph)){
         if(readi(ip, (char*)&ph, off, sizeof(ph)) != sizeof(ph))
             goto bad;
         if(ph.type != ELF_PROG_LOAD)
             continue;
         if(ph.memsz < ph.filesz)
             goto bad;
         if(ph.vaddr + ph.memsz < ph.vaddr)
             goto bad;
         if((sz = allocuvm(pgdir, sz, ph.vaddr + ph.memsz)) == 0)
         //透過allocuvm分配memory給每段ELF
             goto bad;
         if(ph.vaddr % PGSIZE != 0)
             goto bad;
         if(loaduvm(pgdir, (char*)ph.vaddr, ip, ph.off, ph.filesz) < 0)
         //loaduvm->walkpgdir找到要寫入的physical address
             goto bad;
        }
     iunlockput(ip);
     end_op();
     ip = 0;

     // Allocate two pages at the next page boundary.
     // Make the first inaccessible. Use the second as the user stack.
     sz = PGROUNDUP(sz);
     if((sz = allocuvm(pgdir, sz, sz + 2*PGSIZE)) == 0)
         goto bad;
     clearpteu(pgdir, (char*)(sz − 2*PGSIZE));
     sp = sz;

     // Push argument strings, prepare rest of stack in ustack.
     for(argc = 0; argv[argc]; argc++) {
         if(argc >= MAXARG)
             goto bad;
         sp = (sp − (strlen(argv[argc]) + 1)) & ~3;
         if(copyout(pgdir, sp, argv[argc], strlen(argv[argc]) + 1) < 0)
             goto bad;
         ustack[3+argc] = sp;
 }
     ustack[3+argc] = 0;

     ustack[0] = 0xffffffff; // fake return PC
     ustack[1] = argc;
     ustack[2] = sp − (argc+1)*4; // argv pointer

     sp −= (3+argc+1) * 4;
     if(copyout(pgdir, sp, ustack, (3+argc+1)*4) < 0)
         goto bad;

     // Save program name for debugging.
     for(last=s=path; *s; s++)
         if(*s == ’/’)
             last = s+1;
     safestrcpy(curproc−>name, last, sizeof(curproc−>name));

     // install the new image and free the old one
     oldpgdir = curproc−>pgdir;
     curproc−>pgdir = pgdir;
     curproc−>sz = sz;
     curproc−>tf−>eip = elf.entry; // main
     curproc−>tf−>esp = sp;
     switchuvm(curproc);
     freevm(oldpgdir);
     return 0;

    //再分配新memory image的過程中如果有error，會跳到bad，先把新的image free
    //exec要確定新的memory分配會成功，才能free舊的memory
     bad:
         if(pgdir)
             freevm(pgdir);
         if(ip){
             iunlockput(ip);
             end_op();
         }
     return −1;
 }
